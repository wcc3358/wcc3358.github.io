# 1.4 OtaPackageMapper.xml

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.clare.mapper.OtaPackageMapper">

    <!-- 通用查询映射结果 -->
    <resultMap id="BaseResultMap" type="com.clare.model.OtaPackage">
        <id column="package_id" property="packageId" />
        <result column="ota_status" property="otaStatus" />
        <result column="package_name" property="packageName" />
        <result column="package_desc" property="packageDesc" />
        <result column="package_size" property="packageSize" />
        <result column="software_num" property="softwareNum" />
        <result column="current_software_version" property="currentSoftwareVersion" />
        <result column="target_software_version" property="targetSoftwareVersion" />
        <result column="create_at" property="createAt" />
        <result column="update_at" property="updateAt" />
        <result column="create_by" property="createBy" />
        <result column="update_by" property="updateBy" />
    </resultMap>

    <!-- 通用查询结果列 -->
    <sql id="Base_Column_List">
        package_id, ota_status, package_name, package_desc, package_size, software_num, current_software_version, target_software_version, create_at, update_at, create_by, update_by
    </sql>

    <select id="selectOtaPackageList" resultMap="BaseResultMap">
        select * from ota_package
    </select>

</mapper>

# 1.7 MyBatisBootstrap.java

public class MyBatisBootstrap {

    public static void main(String[] args) throws IOException {
        String resource = "mybatis-config.xml";
        InputStream inputStream = Resources.getResourceAsStream(resource);
        SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(inputStream);

        SqlSession sqlSession = sqlSessionFactory.openSession();
        try {
            OtaPackageMapper otaPackageMapper = sqlSession.getMapper(OtaPackageMapper.class);
            List<OtaPackage> otaPackages = otaPackageMapper.selectOtaPackageList();
            for (OtaPackage otaPackage : otaPackages) {
                System.out.println("packageId:"+otaPackage.getPackageId());
            }
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            sqlSession.close();
        }
    }
}
